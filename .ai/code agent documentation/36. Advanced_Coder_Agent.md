```python
def generate_code(self, task: Task, context: CodeContext) -> GeneratedCode:
    """
    Generate code through multi-step pipeline.
    
    Args:
        task: Task to implement
        context: Code context (existing code, style, patterns)
        
    Returns:
        GeneratedCode object with source and metadata
    """
    
    # 1. Analyze task requirements
    requirements = self._analyze_requirements(task)
    
    # 2. Search for similar patterns in codebase
    similar_patterns = context.search_similar(requirements.keywords)
    
    # 3. Generate code
    if similar_patterns:
        # Use hybrid approach: template + refinement
        generated = self._generate_from_patterns(requirements, similar_patterns)
    else:
        # Use pure LLM
        generated = self._generate_with_llm(requirements, context)
    
    # 4. Validate against style guide
    validated = self._validate_style(generated, context)
    
    # 5. Add type hints and docstrings
    enriched = self._add_documentation(validated)
    
    # 6. Return generation result
    return GeneratedCode(
        source=enriched,
        strategy_used=strategy,
        confidence=self._calculate_confidence(requirements),
        issues=[]  # To be filled by self-debugging
    )
```

### 2. Self-Debugging Loop

#### 2.1 Self-Debugging Process

```python
class SelfDebugger:
    """Autonomous code debugging."""
    
    MAX_ITERATIONS = 3
    
    def debug_code(self, generated_code: GeneratedCode, context: CodeContext) -> GeneratedCode:
        """
        Run self-debugging loop.
        
        Args:
            generated_code: Code to debug
            context: Code context with style/patterns
            
        Returns:
            Fixed code or original if no issues found
        """
        current = generated_code
        iteration = 0
        
        while iteration < self.MAX_ITERATIONS:
            # 1. Analyze for issues
            issues = self._find_issues(current, context)
            
            if not issues:
                logger.info("No issues found after debugging")
                return current
            
            logger.info(f"Found {len(issues)} issues in iteration {iteration}")
            
            # 2. Categorize issues
            by_severity = self._categorize_issues(issues)
            
            # 3. Fix issues
            fixed = self._fix_issues(current, by_severity, context)
            
            # 4. Validate fix
            if self._validate_fix(current, fixed, context):
                current = fixed
            else:
                logger.warning("Fix validation failed, reverting")
                break
            
            iteration += 1
        
        return current
```

#### 2.2 Issue Detection

```python
def _find_issues(self, code: GeneratedCode, context: CodeContext) -> List[Issue]:
    """Find potential issues in generated code."""
    issues = []
    
    # 1. Syntax errors
    syntax_issues = self._check_syntax(code.source)
    issues.extend(syntax_issues)
    
    # 2. Type errors
    type_issues = self._check_types(code.source, context)
    issues.extend(type_issues)
    
    # 3. Style violations
    style_issues = self._check_style(code.source, context)
    issues.extend(style_issues)
    
    # 4. Logic errors (static analysis)
    logic_issues = self._check_logic(code.source)
    issues.extend(logic_issues)
    
    # 5. Performance issues
    perf_issues = self._check_performance(code.source)
    issues.extend(perf_issues)
    
    return issues
```

### 3. Quality Assurance

#### 3.1 Quality Metrics

```python
class CodeQuality:
    """Code quality assessment."""
    
    def __init__(self):
        self.correctness: float = 0.0      # Does it work? (0.0-1.0)
        self.readability: float = 0.0      # Is it clear? (0.0-1.0)
        self.maintainability: float = 0.0  # Can it be modified? (0.0-1.0)
        self.efficiency: float = 0.0       # Is it fast enough? (0.0-1.0)
        self.test_coverage: float = 0.0    # Is it tested? (0.0-1.0)
        
    @property
    def overall_score(self) -> float:
        """Calculate overall quality score (0.0-1.0)."""
        return (
            self.correctness * 0.3 +
            self.readability * 0.15 +
            self.maintainability * 0.15 +
            self.efficiency * 0.2 +
            self.test_coverage * 0.2
        )
```

#### 3.2 Quality Gate

```python
def apply_quality_gate(self, generated: GeneratedCode, threshold: float = 0.85) -> bool:
    """
    Apply quality gate - approve/reject code.
    
    Args:
        generated: Generated code with quality metrics
        threshold: Minimum acceptable quality (0.0-1.0)
        
    Returns:
        True if code passes quality gate
    """
    quality = self._assess_quality(generated)
    
    passed = quality.overall_score >= threshold
    
    if passed:
        logger.info(f"Code passed quality gate: {quality.overall_score:.1%}")
    else:
        logger.warning(f"Code failed quality gate: {quality.overall_score:.1%}")
        logger.warning(f"Issues: {quality.get_failing_metrics()}")
    
    return passed
```

### 4. Minimal Diff Generation

#### 4.1 Diff Calculation

```python
class DiffCalculator:
    """Calculate minimal diffs between versions."""
    
    def calculate_diff(self, original: str, modified: str) -> Diff:
        """
        Calculate diff with minimal changes.
        
        Uses unified diff format with context.
        """
        import difflib
        
        original_lines = original.splitlines(keepends=True)
        modified_lines = modified.splitlines(keepends=True)
        
        # Generate unified diff
        diff = difflib.unified_diff(
            original_lines,
            modified_lines,
            lineterm='',
            n=3  # 3 lines of context
        )
        
        diff_lines = list(diff)
        
        # Calculate metrics
        additions = sum(1 for line in diff_lines if line.startswith('+'))
        deletions = sum(1 for line in diff_lines if line.startswith('-'))
        
        return Diff(
            unified='\n'.join(diff_lines),
            additions=additions,
            deletions=deletions,
            total_changes=additions + deletions
        )
```

```python
class CollaborationMessage:
    """Message for agent collaboration."""
    
    def __init__(self, message_type: str, sender: str, receiver: str):
        self.type = message_type  # "request_review", "error_report", etc
        self.sender = sender
        self.receiver = receiver
        self.code = None
        self.context = None
        self.feedback = None
```

#### 5.2 Collaboration Flow

```
CoderAgent generates code
    ↓
Requests review from TesterAgent
    ↓
TesterAgent runs tests
    ├─ Tests pass → Proceed
    └─ Tests fail → Send error report
        ↓
    CoderAgent self-debugs
        ├─ Fixed → Resubmit
        └─ Can't fix → Escalate to SecurityAgent
            ↓
        SecurityAgent checks security
        ├─ OK → Approve
        └─ Issues → Report back
```
